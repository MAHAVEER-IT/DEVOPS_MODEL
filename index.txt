mkdir conflict-demo
cd conflict-demo
git init
echo "version=1.0" > app.txt
echo "message=Hello from main" >> app.txt
git add .
git commit -m "Initial commit"
git branch -M main
git remote add origin <GitHub repo URL>
git push -u origin main
git checkout -b branchA
sed -i 's/Hello from main/Hello from A/' app.txt
git commit -am "Dev A updates message"
git push -u origin branchA
git checkout main
git checkout -b branchB
sed -i 's/Hello from main/Hello from B/' app.txt
git commit -am "Dev B updates message"
git push -u origin branchB
git checkout main
git merge branchA
git push origin main
git merge branchB
